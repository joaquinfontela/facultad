// Hasta acá, (cx,cy) son las coordenadas de la primera intersección

deltay = (pi/2 > alpha) and (alpha > 0) ? p : -p;
deltax = abs(deltay)/tan(alpha);

distanceToWall *= cos(pi/2 - alpha); // Ajustando el "fishbowl effect"
